33  public class XmlStreamWriterTest {
34      /** french */
35      private static final String TEXT_LATIN1 = "eacute: \u00E9";
36  
37      /** greek */
38      private static final String TEXT_LATIN7 = "alpha: \u03B1";
39  
40      /** euro support */
41      private static final String TEXT_LATIN15 = "euro: \u20AC";
42  
43      /** japanese */
44      private static final String TEXT_EUC_JP = "hiragana A: \u3042";
45  
46      /** Unicode: support everything */
47      private static final String TEXT_UNICODE =
48              TEXT_LATIN1 + ", " + TEXT_LATIN7 + ", " + TEXT_LATIN15 + ", " + TEXT_EUC_JP;
49  
50      private static String createXmlContent(String text, String encoding) {
51          String xmlDecl = "<?xml version=\"1.0\"?>";
52          if (encoding != null) {
53              xmlDecl = "<?xml version=\"1.0\" encoding=\"" + encoding + "\"?>";
54          }
55          String xml = xmlDecl + "\n<text>" + text + "</text>";
56          return xml;
57      }
58  
59      private static void checkXmlContent(String xml, String encoding) throws IOException {
60          ByteArrayOutputStream out = new ByteArrayOutputStream();
61          XmlStreamWriter writer = new XmlStreamWriter(out);
62          writer.write(xml);
63          writer.close();
64          byte[] xmlContent = out.toByteArray();
65          String result = new String(xmlContent, encoding);
66          assertEquals(xml, result);
67      }
68  
69      private static void checkXmlWriter(String text, String encoding) throws IOException {
70          String xml = createXmlContent(text, encoding);
71          String effectiveEncoding = (encoding == null) ? "UTF-8" : encoding;
72          checkXmlContent(xml, effectiveEncoding);
73      }

75      /**
76       * <p>testNoXmlHeader.</p>
77       *
78       * @throws java.io.IOException if any.
79       */
80      @Test
81      public void testNoXmlHeader() throws IOException {
82          String xml = "<text>text with no XML header</text>";
83          checkXmlContent(xml, "UTF-8");
84      }

86      /**
87       * <p>testEmpty.</p>
88       *
89       * @throws java.io.IOException if any.
90       */
91      @Test
92      public void testEmpty() throws IOException {
93          ByteArrayOutputStream out = new ByteArrayOutputStream();
94          XmlStreamWriter writer = new XmlStreamWriter(out);
95          writer.flush();
96          writer.write("");
97          writer.flush();
98          writer.write(".");
99          writer.flush();
100         writer.close();
101     }
