42      /** greek */

49      private static final String TEXT_EUC_JP = "hiragana A: \u3042";

51      /** Unicode: support everything */
52      private static final String TEXT_UNICODE =
53              TEXT_LATIN1 + ", " + TEXT_LATIN7 + ", " + TEXT_LATIN15 + ", " + TEXT_EUC_JP;
54  
55      /** see http://unicode.org/faq/utf_bom.html#BOM */
56      private static final byte[] BOM_UTF8 = {(byte) 0xEF, (byte) 0xBB, (byte) 0xBF};
57  
58      private static final byte[] BOM_UTF16BE = {(byte) 0xFE, (byte) 0xFF};
59  
60      private static final byte[] BOM_UTF16LE = {(byte) 0xFF, (byte) 0xFE};
61  
62      private static final byte[] BOM_UTF32BE = {(byte) 0x00, (byte) 0x00, (byte) 0xFF, (byte) 0xFE};
63  
64      private static final byte[] BOM_UTF32LE = {(byte) 0xFF, (byte) 0xFE, (byte) 0x00, (byte) 0x00};

66      private static String createXmlContent(String text, String encoding) {
67          String xmlDecl = "<?xml version=\"1.0\"?>";
68          if (encoding != null) {
69              xmlDecl = "<?xml version=\"1.0\" encoding=\"" + encoding + "\"?>";
70          }
71          String xml = xmlDecl + "\n<text>" + text + "</text>";
72          return xml;
73      }
74  
75      private static void checkXmlContent(String xml, String encoding) throws IOException {
76          checkXmlContent(xml, encoding, null);
77      }
78  
79      private static void checkXmlContent(String xml, String encoding, byte... bom) throws IOException {
80          byte[] xmlContent = xml.getBytes(encoding);
81          InputStream in = new ByteArrayInputStream(xmlContent);
82  
83          if (bom != null) {
84              in = new SequenceInputStream(new ByteArrayInputStream(bom), in);
85          }
86  
87          XmlStreamReader reader = new XmlStreamReader(in);
88          assertEquals(encoding, reader.getEncoding());
89          String result = IOUtil.toString(reader);
90          assertEquals(xml, result);
91      }
92  
93      private static void checkXmlStreamReader(String text, String encoding, String effectiveEncoding)
94              throws IOException {
95          checkXmlStreamReader(text, encoding, effectiveEncoding, null);
96      }
97  
98      private static void checkXmlStreamReader(String text, String encoding) throws IOException {
99          checkXmlStreamReader(text, encoding, encoding, null);
100     }
101 
102     private static void checkXmlStreamReader(String text, String encoding, byte... bom) throws IOException {
103         checkXmlStreamReader(text, encoding, encoding, bom);
104     }
105 
106     private static void checkXmlStreamReader(String text, String encoding, String effectiveEncoding, byte... bom)
107             throws IOException {
108         String xml = createXmlContent(text, encoding);
109         checkXmlContent(xml, effectiveEncoding, bom);
110     }

242     /**
243      * <p>testEncodingAttribute.</p>
244      *
245      * @throws java.io.IOException if any.
246      */
247     @Test
248     public void testEncodingAttribute() throws IOException {
249         String xml = "<?xml version='1.0' encoding='US-ASCII'?><element encoding='attribute value'/>";
250         checkXmlContent(xml, "US-ASCII");
251 
252         xml = "<?xml version='1.0' encoding  =  'US-ASCII'  ?><element encoding='attribute value'/>";
253         checkXmlContent(xml, "US-ASCII");
254 
255         xml = "<?xml version='1.0'?><element encoding='attribute value'/>";
256         checkXmlContent(xml, "UTF-8");
257 
258         xml = "<?xml\nversion='1.0'\nencoding\n=\n'US-ASCII'\n?>\n<element encoding='attribute value'/>";
259         checkXmlContent(xml, "US-ASCII");
260 
261         xml = "<?xml\nversion='1.0'\n?>\n<element encoding='attribute value'/>";
262         checkXmlContent(xml, "UTF-8");
263 
264         xml = "<element encoding='attribute value'/>";
265         checkXmlContent(xml, "UTF-8");
266     }
